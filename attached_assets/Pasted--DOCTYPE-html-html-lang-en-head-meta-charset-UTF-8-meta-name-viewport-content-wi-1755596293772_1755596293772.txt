<script>
  const puppyImages = [
    "https://placedog.net/400/300?id=1",
    "https://placedog.net/400/300?id=2",
    "https://placedog.net/400/300?id=3",
    "https://placedog.net/400/300?id=4",
    "https://placedog.net/400/300?id=5",
    "https://placedog.net/400/300?id=6",
    "https://placedog.net/400/300?id=7",
  ];

  const petNames = ["Buddy", "Bella", "Charlie", "Luna", "Max", "Daisy", "Rocky", "Molly", "Cooper", "Sadie"];

  function randomAge() {
    return (Math.random() * 0.5 + 1.5).toFixed(1) + " years";
  }

  function randomFromArray(arr) {
    return arr[Math.floor(Math.random() * arr.length)];
  }

  const puppyContainer = document.getElementById("puppyContainer");

  puppyImages.forEach((img) => {
    const card = document.createElement("div");
    card.className = "puppy-card";

    const puppyName = randomFromArray(petNames);
    const puppyAge = randomAge();

    card.innerHTML = `
      <img src="${img}" alt="${puppyName}">
      <div class="puppy-name">${puppyName}</div>
      <div class="puppy-info">Age: ${puppyAge}</div>
      <div class="puppy-info">Vaccinated: Yes</div>
      <div class="puppy-info">Insured: Yes</div>
      <div class="puppy-info">Healthy: 100%</div>
      <div class="puppy-info">Price: $3,000 (Negotiable)</div>
      <div class="puppy-info">Delivery: Free in US, Half Price International</div>
      <button class="see-more-btn" onclick="seeMore('${puppyName}', '${puppyAge}', '${img}')">See More</button>
    `;

    puppyContainer.appendChild(card);
  });

  // Modal functionality
  const modal = document.getElementById("puppyModal");
  const modalContent = document.getElementById("modalContent");
  const modalClose = document.getElementById("modalClose");

  function seeMore(name, age, img) {
    modalContent.innerHTML = `
      <img src="${img}" style="width:100%; border-radius:8px; margin-bottom:10px;">
      <h2>${name}</h2>
      <p>Age: ${age}</p>
      <p>Vaccinated: Yes</p>
      <p>Insured: Yes</p>
      <p>Healthy: 100%</p>
      <p>Price: $3,000 (Negotiable)</p>
      <p>Delivery: Free in US, Half Price International</p>
    `;
    modal.style.display = "block";
  }

  modalClose.onclick = function() {
    modal.style.display = "none";
  }

  window.onclick = function(event) {
    if (event.target == modal) {
      modal.style.display = "none";
    }
  }
</script>
